#include <stdio.h>
#include <string.h>


// ETAPA 4 – Struct de contato
struct Contato {
    char nome[50];
    char email[50];
    char tel[30];
};


// ETAPA 5 – Modularização (Funções)


void adicionarContato(struct Contato contatos[], int *total, int limite) {
    if (*total >= limite) {
        printf("\nLimite de contatos atingido!\n");
        return;
    }

    printf("\n--- Adicionar Contato ---\n");

    printf("Digite o nome: ");
    scanf(" %[^\n]", contatos[*total].nome);

    printf("Digite o email: ");
    scanf("%s", contatos[*total].email);

    // ETAPA 2 – Comparação de strings
    
    if (strcmp(contatos[*total].nome, contatos[*total].email) == 0) {
        printf("ATENÇÃO: Nome e email são iguais!\n");
    }

    printf("Digite o telefone: ");
    scanf("%s", contatos[*total].tel);

    (*total)++;
    printf("\nContato adicionado com sucesso!\n");
}

void listarContatos(struct Contato contatos[], int total) {
    if (total == 0) {
        printf("\nNenhum contato para listar!\n");
        return;
    }

    printf("\n=== Lista de Contatos ===\n");
    for (int i = 0; i < total; i++) {
        printf("\nContato %d\n", i + 1);
        printf("Nome: %s\n", contatos[i].nome);
        printf("Email: %s\n", contatos[i].email);
        printf("Telefone: %s\n", contatos[i].tel);
    }
}

void buscarContato(struct Contato contatos[], int total) {
    char nomeBusca[50];

    printf("\nDigite o nome a buscar: ");
    scanf(" %[^\n]", nomeBusca);

    for (int i = 0; i < total; i++) {
        if (strcmp(contatos[i].nome, nomeBusca) == 0) {
            printf("\n=== Contato Encontrado ===\n");
            printf("Nome: %s\n", contatos[i].nome);
            printf("Email: %s\n", contatos[i].email);
            printf("Telefone: %s\n", contatos[i].tel);
            return;
        }
    }

    printf("\nContato nao encontrado!\n");
}

void removerContato(struct Contato contatos[], int *total) {
    char nomeRemover[50];

    printf("\nDigite o nome do contato que deseja remover: ");
    scanf(" %[^\n]", nomeRemover);

    for (int i = 0; i < *total; i++) {
        if (strcmp(contatos[i].nome, nomeRemover) == 0) {
            // Remoção por deslocamento
            for (int j = i; j < (*total) - 1; j++) {
                contatos[j] = contatos[j + 1];
            }
            (*total)--;

            printf("\nContato removido com sucesso!\n");
            return;
        }
    }

    printf("\nContato nao encontrado!\n");
}


// ETAPA 1 – Programa Principal

int main() {
    int limite;
    int total = 0;
    int opcao;

    printf("Digite o número máximo de contatos: ");
    scanf("%d", &limite);

    // ETAPA 3 – Criar array de struct
    struct Contato contatos[limite];

    do {
        printf("\n===== MENU =====\n");
        printf("1 - Adicionar contato\n");
        printf("2 - Listar contatos\n");
        printf("3 - Buscar contato\n");
        printf("4 - Remover contato\n");
        printf("5 - Sair\n");
        printf("Escolha uma opcao: ");
        scanf("%d", &opcao);

        switch (opcao) {
            case 1:
                adicionarContato(contatos, &total, limite);
                break;
            case 2:
                listarContatos(contatos, total);
                break;
            case 3:
                buscarContato(contatos, total);
                break;
            case 4:
                removerContato(contatos, &total);
                break;
            case 5:
                printf("\nFinalizando programa...\n");
                break;
            default:
                printf("Opcao invalida!\n");
        }

    } while (opcao != 5);

    return 0;
}
